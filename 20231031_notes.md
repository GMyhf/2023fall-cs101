#include<bits/stdc++.h>
using namespace std;

int a[50];

long long Cantor(int s[],int n)
{
  bool used[50] = {0};
  long long ans = 0;
  for(int i = n; i >= 1; i--)
  {
    int NoUse = 0;
    used[s[n - i]] = true;
    for(int j = 0; j < s[n - i]; j++)      //search number used
      if(used[j])
        NoUse++;
    ans= (ans + s[n - i] - NoUse) * i;     //computing cantor expansion
  }
  return ans+1;
}

  return 0;
}
